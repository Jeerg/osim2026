// portal/src/graph/foundation/matrix/matrix-clipboard.ts

export const MATRIX_CLIPBOARD_MIME = "application/x-osim-matrix-cells";

export interface MatrixClipboardCell<TVal = unknown> {
  row: number;   // Relative Row-Offset zum Block-Ursprung (analog C++ m_pRPos.y)
  col: number;   // Relative Col-Offset zum Block-Ursprung (analog C++ m_pRPos.x)
  value: TVal;
}

export interface MatrixClipboardPayload<TVal = unknown> {
  origin: string;  // Viewer-Klass des Source-Viewers — z.B. "PRessBelegMatrixViewer"
  cells: MatrixClipboardCell<TVal>[];
}

export function serializeClipboard<TVal>(payload: MatrixClipboardPayload<TVal>): string;
export function deserializeClipboard<TVal>(raw: string): MatrixClipboardPayload<TVal> | null;

export function writeToDataTransfer<TVal>(dt: DataTransfer, payload: MatrixClipboardPayload<TVal>): void;
export function readFromDataTransfer<TVal>(dt: DataTransfer): MatrixClipboardPayload<TVal> | null;

// Cross-Viewer-Type-Check (Pure-Function — Konsumenten entscheidet ob Paste OK)
export function checkOriginCompatible(payloadOrigin: string, targetOrigin: string, compatMap?: Record<string, string[]>): boolean;
```

Default-Compat-Map (z.B. PRessBelegMatrixViewer kann von sich selbst pasten, aber nicht von PRessMengeMatrixViewer):
- Wenn `payloadOrigin === targetOrigin` → true
- Wenn `compatMap[targetOrigin]?.includes(payloadOrigin)` → true
- Sonst false

       export function deserializeClipboard<TVal>(raw: string): MatrixClipboardPayload<TVal> | null {
         if (typeof raw !== "string" || raw.length === 0) return null;
         try {
           const parsed = JSON.parse(raw) as unknown;
           if (typeof parsed !== "object" || parsed === null) return null;
           const p = parsed as Record<string, unknown>;
           if (typeof p.origin !== "string") return null;
           if (!Array.isArray(p.cells)) return null;
           // Cells minimal-validieren
           for (const c of p.cells) {
             if (typeof c !== "object" || c === null) return null;
             const cc = c as Record<string, unknown>;
             if (typeof cc.row !== "number" || typeof cc.col !== "number") return null;
           }
           return p as unknown as MatrixClipboardPayload<TVal>;
         } catch {
           return null;
         }
       }
       ```
