// portal/src/graph/foundation/matrix/useBlockSelection.ts

export interface CellRef { row: number; col: number; }

export interface CellRange {
  startRow: number;
  endRow: number;
  startCol: number;
  endCol: number;
}

// Selektion kann entweder ein Range (Drag/Shift) ODER eine Menge einzelner Cells (Ctrl) sein
export interface BlockSelection {
  range?: CellRange;            // gesetzt bei Drag/Shift
  cells?: Set<string>;          // gesetzt bei Ctrl (Format "row:col")
}

// Pure functions (jsdom-testbar ohne DOM)
export function computeCellRange(start: CellRef, end: CellRef): CellRange;
export function isInRange(cell: CellRef, range: CellRange): boolean;
export function extendSelection(current: CellRange | null, cell: CellRef): CellRange;
export function toggleCellInSelection(current: Set<string>, cell: CellRef): Set<string>;
export function cellKey(cell: CellRef): string;  // "${row}:${col}"

// Hook
export interface UseBlockSelectionResult {
  selection: BlockSelection | null;
  dragStart: CellRef | null;
  handleCellPointerDown: (e: React.PointerEvent, cell: CellRef) => void;
  isSelected: (cell: CellRef) => boolean;
  clear: () => void;
}

export function useBlockSelection(): UseBlockSelectionResult;
```

Behavior-Vertrag:
- Plain Click (kein modifier): startet neuen Drag, Selection wird ein neuer Range mit start=end=cell.
- Pointer-Move während Drag: erweitert Range via computeCellRange(dragStart, currentCell).
- Pointer-Up: schließt Drag ab (dragStart=null), Selection-Range bleibt.
- Shift-Click: extendSelection(current.range, cell).
- Ctrl/Meta-Click: toggleCellInSelection(current.cells, cell) → wechselt in Multi-Cell-Modus.
- ESC: clear() → selection=null, dragStart=null.

       export function cellKey(c: CellRef): string { return `${c.row}:${c.col}`; }

       export function computeCellRange(start: CellRef, end: CellRef): CellRange {
         return {
           startRow: Math.min(start.row, end.row),
           endRow:   Math.max(start.row, end.row),
           startCol: Math.min(start.col, end.col),
           endCol:   Math.max(start.col, end.col),
         };
       }

       export function isInRange(cell: CellRef, r: CellRange): boolean {
         return cell.row >= r.startRow && cell.row <= r.endRow
             && cell.col >= r.startCol && cell.col <= r.endCol;
       }

       export function extendSelection(current: CellRange | null, cell: CellRef): CellRange {
         if (!current) return { startRow: cell.row, endRow: cell.row, startCol: cell.col, endCol: cell.col };
         return computeCellRange({ row: current.startRow, col: current.startCol }, cell);
       }

       export function toggleCellInSelection(current: Set<string>, cell: CellRef): Set<string> {
         const next = new Set(current);
         const k = cellKey(cell);
         if (next.has(k)) next.delete(k); else next.add(k);
         return next;
       }
       ```
