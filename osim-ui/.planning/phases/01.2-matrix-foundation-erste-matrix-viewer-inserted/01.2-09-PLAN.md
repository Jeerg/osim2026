         let modelId: string | null = null;
         try {
           // 1. Login + Upload (analog zu Test 1)
           await loginAs(page, ADMIN.email, ADMIN.password);
           await page.goto("/models");
           await page.getByTestId("btn-upload-otx").click();
           await page.locator("#upload-otx-file").setInputFiles(EMBB_OTX_PATH);
           await page.locator("#upload-otx-name").fill("E2E-Matrix-CopyPaste-" + Date.now());
           await page.getByRole("button", { name: /^Hochladen$/i }).click();
           await page.waitForURL(/\/models\/[a-f0-9-]{36}/, { timeout: 30_000 });
           modelId = page.url().split("/").pop()?.split("?")[0] ?? null;
           expect(modelId).toBeTruthy();

           // 2. Matrix-Viewer öffnen
           await expect(page.locator('[data-viewer="PRessBelegMatrixViewer"]')).toBeVisible({ timeout: 10_000 });

           // 3. Source-Cell A editieren — Status "standard" setzen
           const cells = page.locator('[data-matrix-cell]');
           const cellA = cells.nth(0);
           const cellAId = await cellA.getAttribute("data-matrix-cell");
           await cellA.click();
           await page.locator(`[data-matrix-cell="${cellAId}"][data-matrix-cell-editing="true"] select`).selectOption({ value: "1" });

           // 4. Optional: zweite Source-Cell B editieren mit anderem Status (für echte 2x2-Range)
           const cellB = cells.nth(1);
           const cellBId = await cellB.getAttribute("data-matrix-cell");
           await cellB.click();
           await page.locator(`[data-matrix-cell="${cellBId}"][data-matrix-cell-editing="true"] select`).selectOption({ value: "2" });

           // 5. Block-Select über cellA + cellB:
           //    Click cellA, dann Shift+Click cellB (extends Selection)
           await cellA.click();
           await page.keyboard.down("Shift");
           await cellB.click();
           await page.keyboard.up("Shift");

           // 6. Ctrl+C
           await page.keyboard.press("Control+C");
           // Kurze Pause damit die copy-Event-Schleife durchläuft
           await page.waitForTimeout(100);

           // 7. Target-Cell auswählen (eine andere als cellA/cellB)
           const cellTarget = cells.nth(3);
           const cellTargetId = await cellTarget.getAttribute("data-matrix-cell");
           await cellTarget.click();

           // 8. Ctrl+V
           await page.keyboard.press("Control+V");
           await page.waitForTimeout(500);

           // 9. Assert: cellTarget hat den Wert von cellA (= "standard")
           //    Anchor-Konvention: Paste applied vom Anchor der Range; cellA war der Anchor.
           const cellTargetAfter = page.locator(`[data-matrix-cell="${cellTargetId}"]`);
           await expect(cellTargetAfter).toContainText(/standard/i, { timeout: 5_000 });

           // 10. Wenn die Paste-Implementierung die volle 2x2-Range einsetzt, sollte auch die direkt
           //     benachbarte Cell den Wert von cellB ("notfalls") tragen. Soft-Assert:
           const nextCell = cells.nth(4);
           const nextCellText = (await nextCell.textContent()) ?? "";
           // toleriert beide Implementations-Varianten (single-anchor-Paste ODER full-range-Paste)
           expect(nextCellText.toLowerCase()).toMatch(/notfalls|standard|—/);
         } finally {
           if (modelId) {
             try {
               const token = await getIdToken(page);
               await page.request.delete(`${API_BASE_URL}/api/v1/models/${modelId}`, {
                 headers: { Authorization: `Bearer ${token}` },
               });
             } catch (e) {
               console.error("[cleanup] failed to delete copy/paste test model", e);
             }
           }
         }
       });
