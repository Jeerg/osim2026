**Parser-Pattern** (1:1 aus `matrix-common.tsx` Z.174–186):

```typescript
function parseRaw(raw: string, opts: UseInlineCellEditOptions<unknown>): unknown {
  if (opts.octrlType === "Enum") return Number(raw);
  if (opts.octrlType === "Bool") return Boolean(raw);
  if (opts.valueType === "int") {
    if (raw === "") return 0;
    const p = parseInt(raw, 10);
    return Number.isNaN(p) ? 0 : p;
  }
  if (opts.valueType === "float") {
    if (raw === "") return 0;
    const p = parseFloat(raw);
    return Number.isNaN(p) ? 0 : p;
  }
  return raw;
}
```

### 4. `portal/src/graph/foundation/matrix/useBlockSelection.ts`

**Rolle:** Pure-functions + State-Hook für Drag-Rectangle + Shift/Ctrl-Click.

**Pure-Function-Pattern** (CONTEXT D-5 — wichtig: pure, kein State, jsdom-testbar):

```typescript
export interface CellRange {
  startRow: number;
  endRow: number;
  startCol: number;
  endCol: number;
}

export interface CellRef { row: number; col: number; }

export function computeCellRange(start: CellRef, end: CellRef): CellRange {
  return {
    startRow: Math.min(start.row, end.row),
    endRow: Math.max(start.row, end.row),
    startCol: Math.min(start.col, end.col),
    endCol: Math.max(start.col, end.col),
  };
}

export function isInRange(cell: CellRef, range: CellRange): boolean {
  return (
    cell.row >= range.startRow && cell.row <= range.endRow &&
    cell.col >= range.startCol && cell.col <= range.endCol
  );
}

export function extendSelection(current: CellRange | null, cell: CellRef): CellRange {
  if (!current) return { startRow: cell.row, endRow: cell.row, startCol: cell.col, endCol: cell.col };
  return computeCellRange({ row: current.startRow, col: current.startCol }, cell);
}
```

**Ctrl+Drag-State-Pattern** (1:1 aus `PDurchlaufplanViewerDesign.tsx` Z.417–469 — globale `mousemove`/`mouseup`-Listener mit Cleanup):

```typescript
export function useBlockSelection(): UseBlockSelectionResult {
  const [dragStart, setDragStart] = React.useState<CellRef | null>(null);
  const [selection, setSelection] = React.useState<CellRange | null>(null);

  // Global pointer-listener nur aktiv während Drag (analog PDurchlaufplan Z.418-469)
  React.useEffect(() => {
    if (!dragStart) return;
    const onMove = (e: PointerEvent) => {
      const cell = cellFromEventTarget(e.target);
      if (cell) setSelection(computeCellRange(dragStart, cell));
    };
    const onUp = () => setDragStart(null);
    const onKey = (e: KeyboardEvent) => {
      if (e.key === "Escape") { setDragStart(null); setSelection(null); }
    };
    window.addEventListener("pointermove", onMove);
    window.addEventListener("pointerup", onUp);
    window.addEventListener("keyup", onKey);
    return () => {
      window.removeEventListener("pointermove", onMove);
      window.removeEventListener("pointerup", onUp);
      window.removeEventListener("keyup", onKey);
    };
  }, [dragStart]);
  // ...
}
```

```typescript
  /** Viewer-Klass des Source-Viewers für Type-Compatibility-Check beim Paste. */
  origin: string;
  /** Relativ-Positionen zum Block-Ursprung (analog C++ m_pRPos / PRessBelegMatrix_CellCopyStruct.relpoint). */
```

**Serialize-Pattern** (analog PlanToolbar Z.164–169 — `DataTransfer.setData` + JSON):

```typescript
export function serializeClipboard<TVal>(payload: MatrixClipboardPayload<TVal>): string {
  // structuredClone-Disziplin aus snapshot-service Z.82: niemals Live-Ref weitergeben
  return JSON.stringify({
    origin: payload.origin,
    cells: structuredClone(payload.cells),
  });
}

export function writeToDataTransfer<TVal>(
  dt: DataTransfer,
  payload: MatrixClipboardPayload<TVal>,
): void {
  dt.effectAllowed = "copy";
  dt.setData(MATRIX_CLIPBOARD_MIME, serializeClipboard(payload));
  // Plain-Text-Fallback ausdrücklich NICHT setzen (RESEARCH §Security: kein
  // Plain-Text-Leak in Cross-Origin-Targets).
}
```

**Deserialize-Pattern** (defensiv — analog snapshot-service-Pattern "Malformed-JSON → null"):

```typescript
export function deserializeClipboard<TVal>(raw: string): MatrixClipboardPayload<TVal> | null {
  try {
    const parsed = JSON.parse(raw) as MatrixClipboardPayload<TVal>;
    if (typeof parsed.origin !== "string" || !Array.isArray(parsed.cells)) return null;
    return parsed;
  } catch {
    return null;
  }
}

export function readFromDataTransfer<TVal>(dt: DataTransfer): MatrixClipboardPayload<TVal> | null {
  const raw = dt.getData(MATRIX_CLIPBOARD_MIME);
  if (!raw) return null;
  return deserializeClipboard<TVal>(raw);
}
```

**Cross-Viewer-Origin-Check** (User-Type-Safety): bei Paste mit `payload.origin !== currentViewerKlass` einen Warn-Toast über `sonner` zeigen (Pattern aus `PDurchlaufplanViewerDesign.tsx` Z.174 `toast.error`).

```typescript
    const existing = cellMap.get(`${rowRess.oid}:${colKnoten.oid}`);

    if (newStatus === null) {
      if (!existing) return;
      const newList = (existing.attrs.m_lRessourcen as number[]).filter((oid) => oid !== rowRess.oid);
      if (newList.length === 0) {
        store.deleteObject(existing.oid);
      } else {
        store.patchObject(existing.oid, { m_lRessourcen: newList });
      }
    } else if (existing) {
      store.patchObject(existing.oid, { m_eStatus: newStatus });
    } else {
      // PRBM_OR = Best-Match-Suche; PRBM_AND = neuer PAssozBeleg
      const newOid = store.createObject("PAssozBeleg", {
        m_lRessourcen: [rowRess.oid],
        m_eStatus: newStatus,
      });
      store.appendSubRef(colKnoten.oid, 0, newOid);
    }
    bumpRevision();
  },
  [cellMap, bumpRevision],
);
```
