```typescript
// portal/src/graph/foundation/matrix/useInlineCellEdit.ts

export type OctrlType = "Variable" | "Enum" | "Bool";
export type ValueType = "string" | "int" | "float";

export interface EnumOption {
  value: number;
  label_de: string;
}

export interface UseInlineCellEditOptions<TVal> {
  cellId: string;
  value: TVal | null;
  onCommit: (newValue: TVal) => void;
  onCancel?: () => void;
  readOnly?: boolean;
  octrlType: OctrlType;
  valueType?: ValueType;          // für octrlType === "Variable"
  enumValues?: EnumOption[];       // für octrlType === "Enum"
}

export interface UseInlineCellEditResult {
  editing: boolean;
  startEdit: () => void;
  commit: (raw: string | number | boolean) => void;
  cancel: () => void;
  EditorElement: React.ReactElement | null;  // null wenn !editing
}

export function useInlineCellEdit<TVal>(
  opts: UseInlineCellEditOptions<TVal>,
): UseInlineCellEditResult;

// Pure-Function exportieren für direkten Test:
export function parseRaw(
  raw: unknown,
  octrlType: OctrlType,
  valueType?: ValueType,
): unknown;
```

Parser-Regeln (1:1 aus matrix-common.tsx Z.174-186):
- `octrlType=Enum` → `Number(raw)`
- `octrlType=Bool` → `Boolean(raw)`
- `octrlType=Variable + valueType=int`: "" → 0, parseInt(raw,10), NaN → 0
- `octrlType=Variable + valueType=float`: "" → 0, parseFloat(raw), NaN → 0
- `octrlType=Variable + valueType=string`: raw

       ```typescript
       export function parseRaw(raw: unknown, octrlType: OctrlType, valueType?: ValueType): unknown {
         if (octrlType === "Enum") return Number(raw);
         if (octrlType === "Bool") return Boolean(raw);
         // Variable
         const vt = valueType ?? "string";
         if (vt === "int") {
           if (raw === "" || raw == null) return 0;
           const p = parseInt(String(raw), 10);
           return Number.isNaN(p) ? 0 : p;
         }
         if (vt === "float") {
           if (raw === "" || raw == null) return 0;
           const p = parseFloat(String(raw));
           return Number.isNaN(p) ? 0 : p;
         }
         return raw;
       }
       ```
