           function buildKnotenDetailGrid(knoten: OBaseObj, allObjects: Record<number, OBaseObj>): OGraphGrid {
             const grid = new OGraphGrid();
             // 1. Knoten zentral bei (3, 0), forbidden move + delete
             const knotenGObj = new GObjLink();
             knotenGObj.SetViewedObject(`oid:${knoten.oid}` as never);
             knotenGObj.SetText(((knoten.attrs.m_sName as string | undefined) ?? `oid ${knoten.oid}`));
             knotenGObj.SetMoveForbidden(true);
             knotenGObj.SetDeleteForbidden(true);
             grid.GOIns(knotenGObj, 3, 0, false);

             // 2. Assoz-Ressourcen aus m_lAssozRess (attrs ODER sub_refs[<SLOT>])
             const assozRessOids = (knoten.attrs.m_lAssozRess as number[] | undefined) ?? knoten.sub_refs[<SLOT_FROM_AUDIT>] ?? [];
             for (const aOid of assozRessOids) {
               const assoz = allObjects[aOid];
               if (!assoz) continue;
               const aGObj = new GObjLink();
               aGObj.SetViewedObject(`oid:${assoz.oid}` as never);
               aGObj.SetText(((assoz.attrs.m_sName as string | undefined) ?? assoz.klass));
               // Position: m_pntRaster wenn vorhanden, sonst NextFreePlace
               const pt = assoz.attrs.m_pntRaster as { col: number; row: number } | undefined;
               if (pt && !grid.IsGridPlaceTaken(pt.col, pt.row)) {
                 grid.GOIns(aGObj, pt.col, pt.row, false);
               } else {
                 const free = grid.GetNextFreeGridPlace();
                 grid.GOIns(aGObj, free.col, free.row, false);
               }
               // 2a. Iteriere PAssozBeleg.m_lRessourcen → Ressourcen-Nodes + Links Assoz→Ress
               const ressOids = (assoz.attrs.m_lRessourcen as number[] | undefined) ?? assoz.sub_refs[<SLOT>] ?? [];
               for (const rOid of ressOids) {
                 const ress = allObjects[rOid];
                 if (!ress) continue;
                 const rGObj = new GObjLink();
                 rGObj.SetViewedObject(`oid:${ress.oid}` as never);
                 rGObj.SetText(((ress.attrs.m_sName as string | undefined) ?? ress.klass));
                 const rFree = grid.GetNextFreeGridPlace();
                 grid.GOIns(rGObj, rFree.col, rFree.row, false);
                 // Link AssozRess → Ressource
                 grid.addGLink(aGObj, rGObj);  // ggf. ist die API "addLink" / new GLink(...) — siehe OGraphGrid.ts
               }
               // Link Knoten → AssozRess
               grid.addGLink(knotenGObj, aGObj);

             const assozSpeichOids = (knoten.attrs.m_lAssozSpeich as number[] | undefined) ?? [];
             for (const sOid of assozSpeichOids) { /* analog */ }

             return grid;
