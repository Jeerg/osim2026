```typescript
// portal/src/api/runs.ts
export interface RunSubmitRequest {
  model_id: string;
  seed: number;
  periods: number;
  timeout_seconds?: number;
}

export type RunStatus = 'queued' | 'running' | 'succeeded' | 'failed' | 'timeout';

export interface RunResponse {
  run_id: string;
  status: RunStatus;
  started_at: string | null;
  finished_at: string | null;
  summary: {
    event_count: number;
    duration_s: number;
    counters: Record<string, number>;
    topics: Record<string, number>;
  } | null;
  error: string | null;
}

export interface TraceResponse {
  run_id: string;
  trace_url: string;     // signierter Download-URL
  expires_at: string;
  size_bytes: number;
}

export function createRun(req: RunSubmitRequest): Promise<RunResponse>;
export function getRun(runId: string): Promise<RunResponse>;
export function listRuns(limit?: number): Promise<RunResponse[]>;
export function getRunTrace(runId: string): Promise<TraceResponse>;

// portal/src/hooks/useRunStatusPolling.ts
export function useRunStatusPolling(runId: string, opts?: {
  intervalMs?: number;    // Default 2000
  enabled?: boolean;
}): {
  run: RunResponse | undefined;
  isLoading: boolean;
  isTerminal: boolean;
  error: Error | null;
};
```
