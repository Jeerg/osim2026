# KPI-Kategorien (aus OSim2004 § 5):

class KPIValue:
  value: float | int
  unit: str        # "h", "%", "Stück", "EUR", "EUR/h"
  description: str # "Mittlere Durchlaufzeit über alle Auslösungen"

class DurchlaufplanKPIs:
  anzahl_fertiggestellt: KPIValue
  mittlere_durchlaufzeit: KPIValue
  minimale_durchlaufzeit: KPIValue
  zielerreichungsgrad_durchlaufzeit: KPIValue | None  # nur wenn Zielzeit gesetzt
  prozesskosten_gesamt: KPIValue
  prozesskosten_pro_stueck: KPIValue

class KnotenKPIs(DurchlaufplanKPIs):
  knoten_id: str
  knoten_name: str

class BelegungsRessourceKPIs:
  theor_kapazitaetsbestand: KPIValue
  abgearb_kapazitaetsbedarf: KPIValue
  auslastung: KPIValue              # Prozent
  einsatzzeit: KPIValue              # Stunden
  perioden_kosten: KPIValue
  kostensatz: KPIValue               # EUR/h
  zeitstressgrad: KPIValue
  zielerreichungsgrad_ermuedung: KPIValue | None

class MengenRessourceKPIs:
  ressource_id: str
  bestand_min: KPIValue
  bestand_max: KPIValue
  bestand_mittel: KPIValue
  verbrauch_gesamt: KPIValue

class RunKPIReport:
  run_id: UUID
  computed_at: datetime
  durchlaufplaene: list[DurchlaufplanKPIs]
  knoten: list[KnotenKPIs]
  belegungs_ressourcen: list[BelegungsRessourceKPIs]
  mengen_ressourcen: list[MengenRessourceKPIs]

class AggregatedKPI:
  mean: float
  median: float
  min: float
  max: float
  std: float
  ci95_low: float
  ci95_high: float
  n_runs: int
</interfaces>
