export type Oid = number;
export type Klass = string;

export interface OtxJsonNode {
  oid: Oid;
  klass: Klass;
  name: string;
  properties: Record<string, string | number | boolean | null>;
  children: OtxJsonNode[];
  unsupported?: true;
}

export interface ViewerRegistration {
  klass: Klass;
  component: ChildDialogComponent;
  displayName: string;
  // optional: priority (für Mehrfach-Registrierung später)
}

// portal/src/viewers/core/ViewerFrame.ts (Klasse, kein React)
export class ViewerFrame {
  constructor(public title: string, public onObjChange?: (obj: OtxJsonNode) => void) {}
  setObj(obj: OtxJsonNode): void;       // delegiert an ClientCtrl
  update(): void;                       // re-fetches state, triggert React-Re-Render
  getMenu(): ViewerMenuSpec | null;     // null = kein Menu
}

// portal/src/viewers/core/ClientCtrl.ts (Klasse)
export class ClientCtrl {
  current: OtxJsonNode | null = null;
  childDialogKlass: Klass | null = null;  // welcher Dialog gerade gemounted ist
  constructor(public frame: ViewerFrame) {}
  setObj(obj: OtxJsonNode): boolean;    // OViewer.h 3.3: prüft store, mountet ggf. neuen ChildDialog
  store(): boolean;                     // ChildDialog flusht zu Store; false = nicht möglich
  update(): void;                       // re-render-Signal
  pickChildDialog(klass: Klass): ChildDialogComponent | null;  // delegiert an registry
}

// portal/src/viewers/core/ChildDialog.tsx
export interface ChildDialogProps {
  obj: OtxJsonNode;
  onPropertyChange: (oid: Oid, key: string, value: unknown) => void;
  onMethodCall: (oid: Oid, method: string, args?: unknown[]) => void;
  // weitere Standard-Hooks
}
export type ChildDialogComponent = React.FC<ChildDialogProps>;

// portal/src/viewers/core/ChildCtrl.tsx — React-Wrapper für nested ChildDialog (analog C++-OCtrlViewChild)
export interface ChildCtrlProps {
  obj: OtxJsonNode;                     // sub-Objekt
  // Routing-Forwarding-Defaults
}

// portal/src/viewers/core/OCtrl.types.ts
export interface OCtrlProps<T = unknown> {
  property: string;                     // m_sName, m_iBeginTermin, ...
  label?: string;
  readonly?: boolean;
}
export function useOCtrlBinding<T>(prop: string): {
  value: T;
  setValue: (v: T) => void;
  metadata: OCtrlMetadata;              // type, enum-values, link-target-klass etc. (aus Type-Map)
};

// portal/src/state/model-store.ts (Zustand)

  addChild(parentOid: Oid, newNode: OtxJsonNode): void;
  removeNode(oid: Oid): void;
  selectOid(oid: Oid | null): void;
  undo(): void;
  redo(): void;
  markClean(): void;
}
export const useModelStore: UseBoundStore<StoreApi<ModelState>>;

<risks>
- **OViewer-Routing-Komplexität:** OViewer.h beschreibt 4 verschiedene Routing-Pfade (Kommando, Update, Set-Obj, Store). Phase 1 deckt davon ab: Set-Obj (über ClientCtrl.setObj) und Update (über React-Re-Render). Kommando-Routing (Menü/Toolbar) und Store-Routing-Validation kommen später. Risiko: spätere Konsumenten erwarten Kommando-Routing — Plan 04 dokumentiert in SUMMARY.md, was NICHT implementiert ist, damit Plan 05+ kein Vertrauen darauf aufbaut.
- **Tree-Walk-Performance:** Bei großen Modellen (Bosch2_wechseln 18MB) wird der oid-Index essenziell. Plan 04 implementiert oid-Map-Cache schon korrekt (rebuild on setTree).
- **Snapshot-Undo bei 18MB-Modellen:** JSON.stringify(tree) bei jedem Edit = teuer. Phase-1-Mitigation: undoStack-Limit 50, sample-frequency 1 pro Property-Change ist OK für Dummy/Fertigungsstruktur1; Bosch2 wird in Phase 1 nur als Read-Test gefahren (Edit-Performance kein Acceptance-Criterion). Phase 3+ ersetzt durch Command-Pattern.
- **OCtrlLink + OCtrlList brauchen Cross-Tree-Lookups:** Beim Rendern eines Link-Dropdowns für "Knoten → PRessBeleg" muss das Model die PRessBeleg-Liste kennen. Plan 04 macht useModelStore mit einem candidates-by-klass-Selektor verfügbar (Memoized über Map<Klass, OtxJsonNode[]>).
- **Stale-State im ClientCtrl bei React-Strict-Mode:** TS-Klasse + React-State haben kein automatisches Sync. Mitigation: ClientCtrl-Subscribe-Mechanik (notify-Listener), ViewerHost via useEffect+useState aufgehängt.
</risks>
