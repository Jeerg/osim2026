       ```typescript
       export interface RecoveryCheck {
         hasSnapshot: boolean;
         snapshotIsNewer: boolean;
         snapshotSavedAt?: number;
         snapshotVersion?: number;
         serverVersion: number;
       }
       
       export async function checkForRecovery(modelId: number, serverVersion: number): Promise<RecoveryCheck> {
         const snap = await readLatestSnapshot(modelId);
         if (!snap) return { hasSnapshot: false, snapshotIsNewer: false, serverVersion };
         return {
           hasSnapshot: true,
           snapshotIsNewer: snap.dirty.length > 0 || snap.version < serverVersion,  // dirty UND/ODER alte Version-Basis
           snapshotSavedAt: snap.savedAt,
           snapshotVersion: snap.version,
           serverVersion,
         };
       }
       ```

       export function useSnapshotSubscriber(modelId: number) {
         const writeDebounced = useMemo(() => debounce((tree, dirty, version) => {
           writeSnapshot(modelId, version, tree, dirty);
         }, 500), [modelId]);
         
         useEffect(() => {
           const unsub = useModelStore.subscribe((state) => {
             if (state.tree && state.modelId === modelId) {
               writeDebounced(state.tree, state.dirty, state.version!);
             }
           });
           return unsub;
         }, [modelId, writeDebounced]);
       }

<risks>
- **IndexedDB-Quota bei vielen Modellen:** Snapshot pro Modell, ~ein paar MB pro Snapshot bei Dummy.otx-Größe, multipliziert mit Anzahl offener Modelle → kann Quota sprengen. Mitigation: alte Snapshots eviction (keep last 5 modelIds, FIFO). Phase 1: nicht implementieren, dokumentieren als Backlog.
- **oid_mapping-Patching kann Bugs verursachen:** wenn eine OID in einer property als string statt number vorliegt (z.B. m_lVon: "5" statt 5), wird sie nicht ersetzt. Mitigation: model-store + JSON-Tree-Service halten OIDs strikt als number. Tests decken nested-references ab.
- **Auto-Save-Conflict mit Manual-Save:** wenn 30s-Timer feuert während Manual-Save läuft → race-condition. Mitigation: ein `saving: boolean`-Flag im model-store; useAutoSave skipt run wenn saving=true.
- **Lock-Heartbeat 60s + Lock-TTL 15min:** großzügiger Sicherheitspuffer (14 Heartbeats in einer TTL-Periode). Bei Netzwerk-Glitch (max 14 verloren) bleibt der Lock noch gehalten.
- **Recovery vs. version-mismatch:** Wenn snapshot.version=3 und server.version=5 (zwischenzeitlich von anderem User saved), kann der Recovery-Tree veraltet sein und Konflikte erzeugen. Phase-1-Strategie: Recovery-Prompt erweitern um Warnung "Server hat zwischenzeitlich Version 5; deine Änderungen basieren auf Version 3 — Konflikt-Auflösung manuell"; "Wiederherstellen" arbeitet trotzdem darauf weiter, beim nächsten Save erzeugt es eine neue Version 6 (kein merge). Conflict-Merge ist Phase 4+ Thema.
</risks>
