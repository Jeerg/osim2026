  getCellValue: (row: RowT, col: ColT) => CellValueT; // Lookup

  renderCell?: (value: CellValueT) => ReactNode;      // default: typeof-basiert
  rowHeader: (row: RowT) => ReactNode;
  colHeader: (col: ColT) => ReactNode;
  rowKey: (row: RowT) => string | number;
  colKey: (col: ColT) => string | number;
  readonly?: boolean;
  virtualized?: boolean;                              // default true für > 30x30
}
