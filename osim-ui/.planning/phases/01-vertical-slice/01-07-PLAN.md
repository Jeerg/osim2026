```typescript
export interface Position { x: number; y: number; }
export interface Size { width: number; height: number; }
export interface Rect extends Position, Size {}

// portal/src/graph/core/GObject.ts
export abstract class GObject {
  constructor(
    public id: string,
    public klass: string,
    public position: Position,
    public size: Size = { width: 120, height: 60 },
  ) {}
  
  abstract render(): ReactNode;
  
  contains(p: Position): boolean { /* default rect-check */ }
  regionCheck(p: Position): 'inside' | 'edge' | 'outside' { /* phase-1: inside|outside binary */ }
  updatePosition(p: Position): void { this.position = p; }
}

// portal/src/graph/core/GLink.ts
export abstract class GLink {
  constructor(
    public id: string,
    public source: string,        // GObject.id
    public target: string,
    public waypoints: Position[] = [],
  ) {}
  abstract render(): ReactNode;
}

// portal/src/graph/core/GObjLink.ts — typed wrapper
export class GObjLink extends GLink {
  constructor(id: string, public sourceObj: GObject, public targetObj: GObject) {
    super(id, sourceObj.id, targetObj.id);
  }
}
```

       ```typescript
       import { ReactFlow, Node, Edge, Background, Controls, ... } from 'reactflow';
       import 'reactflow/dist/style.css';
       
       export const GraphView: FC<GraphViewProps> = ({ objects, links, onObjectMove, onLinkCreate, onObjectDoubleClick, readonly }) => {
         const nodes: Node[] = useMemo(() => objects.map(o => ({
           id: o.id,
           type: 'gobject',
           position: o.position,
           data: { gobj: o, onDoubleClick: () => onObjectDoubleClick?.(o.id) },
         })), [objects, onObjectDoubleClick]);
         
         const edges: Edge[] = useMemo(() => links.map(l => ({
           id: l.id, source: l.source, target: l.target, type: 'kante', data: { glink: l },
         })), [links]);
         
         const nodeTypes = useMemo(() => ({ gobject: GObjectNodeWrapper }), []);
         const edgeTypes = useMemo(() => ({ kante: KanteEdgeWrapper }), []);
         
         const onNodesChange = useCallback((changes) => {
           changes.forEach(c => {
             if (c.type === 'position' && c.position && !c.dragging) {
               onObjectMove?.(c.id, c.position);
             }
           });
         }, [onObjectMove]);
         
         const onConnect = useCallback((conn) => {
           if (conn.source && conn.target) onLinkCreate?.(conn.source, conn.target);
         }, [onLinkCreate]);
         
         return (
           <div style={{ width: '100%', height: '100%' }}>
             <ReactFlow
               nodes={nodes} edges={edges}
               nodeTypes={nodeTypes} edgeTypes={edgeTypes}
               onNodesChange={onNodesChange} onConnect={onConnect}
               nodesDraggable={!readonly} nodesConnectable={!readonly}
               fitView
             >
               <Background />
               <Controls />
             </ReactFlow>
           </div>
         );
       };
       ```

       ```typescript
       export const PDurchlaufplanViewerDesign: ChildDialogComponent = ({ obj, onPropertyChange }) => {
         const updateProperty = useModelStore(s => s.updateProperty);

         const selectOid = useModelStore(s => s.selectOid);
         
         // Knoten aus obj.children extrahieren (klass-filter)

         const gObjects = useMemo(() => knoten.map(k => new KnotenNode(

           { x: Number(k.properties.m_xGfxPosX ?? 0), y: Number(k.properties.m_xGfxPosY ?? 0) },
           String(k.properties.m_sName ?? 'Knoten'),
           Number(k.properties.m_iDurchfuehrungszeit ?? 0),
         )), [knoten]);
         
         const gLinks = useMemo(() => kanten.map(e => new GObjLink(
           String(e.oid),

                 updateProperty(Number(id), 'm_xGfxPosY', pos.y);
               }}
               onLinkCreate={(source, target) => {
       ```

       Option C: Auto-Layout (dagre via reactflow) bei jedem Open. Layout ist deterministisch, nicht persistiert.
