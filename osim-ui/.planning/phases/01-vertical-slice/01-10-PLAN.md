       ```python
       async def test_upload_edit_save_download_roundtrip(app_client, mock_firebase_token, dummy_otx_bytes):
           token = mock_firebase_token(uid="test-user-1", email="test@example.com")
           hdrs = {"Authorization": f"Bearer {token}"}
           
           # Step 1: Auth/me bootstraps tenant
           r = await app_client.post("/api/v1/auth/me", headers=hdrs)
           assert r.status_code == 200
           tenant_id = r.json()["tenant_id"]
           
           # Step 2: Upload OTX
           files = {"file": ("Dummy.otx", dummy_otx_bytes, "application/octet-stream")}
           r = await app_client.post("/api/v1/models/upload-otx", headers=hdrs, files=files)
           assert r.status_code == 200
           model_id = r.json()["id"]
           assert r.json()["coverage_ratio"] > 0
           
           # Step 3: Acquire lock
           r = await app_client.post(f"/api/v1/models/{model_id}/lock", headers=hdrs)
           assert r.status_code == 200
           
           # Step 4: GET tree
           r = await app_client.get(f"/api/v1/models/{model_id}/tree", headers=hdrs)
           assert r.status_code == 200
           tree = r.json()["tree"]
           assert tree["klass"] == "ASimulator"  # oder das, was Plan 03 als Root setzt
           
           # Step 5: Edit (z.B. m_keim ändern)
           tree["properties"]["m_keim"] = 99999.0
           r = await app_client.put(f"/api/v1/models/{model_id}/tree",
                                     headers=hdrs, json={"tree": tree, "expected_version": 1})
           assert r.status_code == 200
           new_version = r.json()["version"]
           assert new_version == 2
           
           # Step 6: GET tree erneut → Edit ist persistiert
           r = await app_client.get(f"/api/v1/models/{model_id}/tree", headers=hdrs)
           assert r.json()["tree"]["properties"]["m_keim"] == 99999.0
           
           # Step 7: Verify Version 1 (Original) unverändert
           r = await app_client.get(f"/api/v1/models/{model_id}/download-original", headers=hdrs)
           # 302-Redirect zu signed url; fetch the redirect
           # ... assert bytes-Identität mit dummy_otx_bytes
       ```
